--- FILE: backend/src/models/types.ts ---
export interface Player {
  id: string;             // Уникальный ID (или Telegram ID)
  name: string;
  telegram: string;
  clutchPoints: number;   // Текущий баланс очков для ставок
  totalEarned: number;    // Всего заработано (метрика эффективности)
  totalSpent: number;     // Всего сожжено на аукционах
  maxStreak: number;      // Лучшая серия побед
  currentStreak: number;  // Текущая серия (для триггера Гран-при)
}

export interface Match {
  id: string;
  playerAId: string;
  playerBId: string;
  scoreA: number;
  scoreB: number;
  winnerId: string;
  bidPool: number;        // Сколько всего поинтов было на кону в этом матче
  grandPrizeContribution: number; // 50% от ставки, ушедшие в пул Гран-при
  timestamp: number;
}

export interface TournamentState {
  currentKingId: string | null;  // Кто сейчас на троне
  grandPrizePool: number;        // Текущий размер пула Гран-при
  queue: { playerId: string; bid: number }[]; // Очередь претендентов, отсортированная по bid
}

// 1. POST /api/auction/bid — Игрок тратит X поинтов. 
//    - Списываем из его clutchPoints, добавляем в totalSpent.
//    - Помещаем в state.queue. Сортируем очередь по убыванию bid.
//
// 2. POST /api/matches/resolve — Фиксация результата матча (Король vs Первый в очереди)
//    - Претендент извлекается из топа очереди: queue.shift()
//    - Общий пул матча = bid претендента.
//    - В фонд Гран-при уходит: grandPrize = Math.floor(bid * 0.5)
//    - Победителю матча начисляется: reward = bid - grandPrize
//    - Обновляем state.grandPrizePool += grandPrize
//    - IF (Победитель == Текущий Король):
//        currentStreak++
//        totalEarned += reward
//        if (currentStreak === 3 || currentStreak === 5) {
//           // Вскрываем Джекпот Гран-при!
//           winner.clutchPoints += state.grandPrizePool;
//           winner.totalEarned += state.grandPrizePool;
//           state.grandPrizePool = 0; // Сброс пула
//        }
//      ELSE (Претендент победил Короля):
//        - Старый король теряет трон, его currentStreak = 0.
//        - Новый король (претендент) садится на трон, currentStreak = 1.
//        - Новый король получает свой reward за этот конкретный матч.
